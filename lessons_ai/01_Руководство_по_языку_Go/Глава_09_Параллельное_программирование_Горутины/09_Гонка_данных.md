    "fmt"

type SafeCounter struct {
    mu    sync.Mutex
    value int
}

    c.mu.Lock()
    defer c.mu.Unlock()
    c.value++

    c.mu.Lock()
    defer c.mu.Unlock()

    c.mu.Lock()
    defer c.mu.Unlock()
    return c.value
