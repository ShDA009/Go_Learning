### –ó–∞–¥–∞–Ω–∏–µ 4: Pipeline
–°–æ–∑–¥–∞–π—Ç–µ –∫–æ–Ω–≤–µ–π–µ—Ä –æ–±—Ä–∞–±–æ—Ç–∫–∏ —á–∏—Å–µ–ª.

package main

import "fmt"

func pipeline(input int, steps ...func(int) int) int {
    result := input
    for _, step := range steps {
        result = step(result)
    }
    return result
}

func main() {
    result := pipeline(5,
        func(x int) int { return x * 2 },  // 10
        func(x int) int { return x + 3 },  // 13
        func(x int) int { return x * x },  // 169
    )
    fmt.Println("–†–µ–∑—É–ª—å—Ç–∞—Ç:", result)
}
