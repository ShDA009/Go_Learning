    "fmt"
    "os"

type Article struct {
    Title   string
    Content string
    Author  string
}

func generateHTML(w *os.File, article Article) {
    fmt.Fprintln(w, "<!DOCTYPE html>")
    fmt.Fprintln(w, "<html>")
    fmt.Fprintln(w, "<head>")
    fmt.Fprintf(w, "    <title>%s</title>\n", article.Title)
    fmt.Fprintln(w, "</head>")
    fmt.Fprintln(w, "<body>")
    fmt.Fprintf(w, "    <h1>%s</h1>\n", article.Title)
    fmt.Fprintf(w, "    <p>%s</p>\n", article.Content)
    fmt.Fprintf(w, "    <footer>By %s</footer>\n", article.Author)
    fmt.Fprintln(w, "</body>")
    fmt.Fprintln(w, "</html>")
}

func main() {
    article := Article{
        Title:   "Hello Go",
        Content: "Go is a great language!",
        Author:  "Gopher",
    }
    
    file, _ := os.Create("article.html")
    defer file.Close()
    
    generateHTML(file, article)
    fmt.Println("HTML generated!")
}
