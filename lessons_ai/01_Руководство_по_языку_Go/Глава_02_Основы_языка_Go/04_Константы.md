### –ó–∞–¥–∞–Ω–∏–µ 4: –ë–∏—Ç–æ–≤—ã–µ —Ñ–ª–∞–≥–∏ –ø—Ä–∞–≤ –¥–æ—Å—Ç—É–ø–∞

```go
    fmt.Printf("Read+Write: %d (binary: %03b)\n", readWrite, readWrite)
    fmt.Printf("Full access: %d (binary: %03b)\n", fullAccess, fullAccess)
    
    // –ü—Ä–æ–≤–µ—Ä–∫–∞ –ø—Ä–∞–≤

    hasRead := permissions & Read != 0
    hasExecute := permissions & Execute != 0

}
```
