package main

import "fmt"

type Point struct {
    X int
    Y int
}

func main() {
    p := Point{10, 20}
    
    fmt.Printf("%%v:  %v\n", p)   // {10 20}

    fmt.Printf("%%#v: %#v\n", p)  // main.Point{X:10, Y:20}
    fmt.Printf("%%T:  %T\n", p)   // main.Point
}
